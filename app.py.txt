    <script>
        // KELİME LİSTESİ VE TÜRKÇE OKUNUŞ REHBERLERİ
        const wordList = {
            "the": "dı / dıı",
            "join": "coyn",
            "jump": "camp",
            "who": "huu",
            "are": "aa",
            "think": "fink (dişlerin arasından s sızdırarak)",
            "thought": "foot (dişlerin arasından f/t sızdırarak)",
            "about": "ıbaut",
            "refuse": "rifyuz",
            "use": "yuz",
            "she": "şii",
            "chat": "çet",
            "accept": "eksept",
            "language": "lenguiç",
            "umbrella": "ambrella",
            "quick": "kuik",
            "what": "vat",
            "where": "vee",
            "three": "trii (dişlerin arasından t sızdırarak)",
            "speak": "spiik",
            "sign": "sayn",
            "location": "lokeyşın",
            "bathroom": "baatruum",
            "today": "tudey",
            "wednesday": "venzdey",
            "thursday": "förzdey",
            "watch": "voç",
            "rarely": "reerli",
            "usually": "yujıli",
            "generally": "cenırli",
            "currently": "karantli",
            "university": "yunivörsiti",
            "choose": "çuuz",
            "country": "kantri"
        };

        // Alfabetik Sıralama Yapalım
        const sortedWords = Object.keys(wordList).sort();

        // Hafıza Durumları (Sayfa yenilendiğinde tamamen uçar)
        let currentWord = sortedWords[0];
        let totalStars = 0;
        let completedWords = new Set();
        
        // Ses Kayıt Değişkenleri
        let mediaRecorder;
        let audioChunks = [];
        let isRecording = false;

        // Tureng Basılı Tutma Zamanlayıcı Değişkenleri
        let pressTimer;
        let isHolding = false;

        // Sayfa Yüklendiğinde Başlat
        window.onload = function() {
            const selectEl = document.getElementById('word-select');
            
            // Seçenek listesini doldur
            sortedWords.forEach(word => {
                const opt = document.createElement('option');
                opt.value = word;
                opt.innerText = word;
                selectEl.appendChild(opt);
            });

            // İlk Kelimeyi Kur
            selectWord(currentWord);

            // Tureng Basılı Tutma Dinleyicileri (Masaüstü ve Mobil için)
            const card = document.getElementById('word-card');
            
            // Mouse Olayları
            card.addEventListener('mousedown', startPress);
            card.addEventListener('mouseup', cancelPress);
            card.addEventListener('mouseleave', cancelPress);

            // Dokunmatik Ekran (Mobil) Olayları
            card.addEventListener('touchstart', function(e) {
                e.preventDefault(); // Sürükleme veya büyüteç açma davranışlarını engelle
                startPress();
            });
            card.addEventListener('touchend', cancelPress);
            card.addEventListener('touchcancel', cancelPress);
        };

        // Tema Seçimi Değiştirici
        function changeTheme(theme) {
            const body = document.getElementById('body-bg');
            body.className = "min-h-screen transition-all duration-500 pb-12";
            if (theme === 'blue') body.classList.add('bg-pastel-blue');
            if (theme === 'pink') body.classList.add('bg-pastel-pink');
            if (theme === 'green') body.classList.add('bg-pastel-green');
            if (theme === 'yellow') body.classList.add('bg-pastel-yellow');
        }

        // Kelime Değiştirme Tetikleyicisi
        function selectWord(word) {
            currentWord = word;
            document.getElementById('display-word').innerText = word;
            document.getElementById('display-hint').innerText = wordList[word];
            document.getElementById('word-select').value = word;

            // Ses kayıt alanını gizle/sıfırla
            document.getElementById('audio-playback-container').classList.add('hidden');
            document.getElementById('audio-playback').src = '';
            document.getElementById('record-status').innerText = "Kaydetmek için kırmızı mikrofona tıklatın";
            document.getElementById('mic-icon').className = "fa-solid fa-microphone text-2xl";
            isRecording = false;

            // Başardım butonunu güncelle
            updateSuccessButtonState();
        }

        // Web Speech API ile TTS Doğru Ses Telaffuzu Dinleme
        function playTTS(rate) {
            // Eğer tarayıcıda konuşma desteği varsa
            if ('speechSynthesis' in window) {
                // Aktif konuşmaları iptal et
                window.speechSynthesis.cancel();

                const utterance = new SpeechSynthesisUtterance(currentWord);
                utterance.lang = 'en-US'; // Amerikan İngilizcesi
                utterance.rate = rate;    // Konuşma hızı

                window.speechSynthesis.speak(utterance);
            } else {
                alert("Üzgünüz, tarayıcınız konuşma sentezleme özelliğini desteklemiyor.");
            }
        }

        // SES KAYDETME FONKSİYONLARI (Web Audio API)
        async function toggleRecording() {
            const statusText = document.getElementById('record-status');
            const micIcon = document.getElementById('mic-icon');

            if (!isRecording) {
                // Kayıt Başlatılıyor
                audioChunks = [];
                try {
                    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
                    mediaRecorder = new MediaRecorder(stream);
                    
                    mediaRecorder.ondataavailable = event => {
                        audioChunks.push(event.data);
                    };

                    mediaRecorder.onstop = () => {
                        const audioBlob = new Blob(audioChunks, { type: 'audio/wav' });
                        const audioUrl = URL.createObjectURL(audioBlob);
                        
                        const playback = document.getElementById('audio-playback');
                        playback.src = audioUrl;
                        document.getElementById('audio-playback-container').classList.remove('hidden');
                    };

                    mediaRecorder.start();
                    isRecording = true;
                    statusText.innerText = "Konuşun... Kayıt yapılıyor... Durdurmak için tekrar basın.";
                    statusText.classList.add('text-red-500', 'font-bold');
                    micIcon.className = "fa-solid fa-square text-2xl animate-pulse"; // Durdurma simgesi
                } catch (err) {
                    console.log("Mikrofon izni alınamadı veya mikrofon bulunamadı.", err);
                    statusText.innerText = "Mikrofon izni verilmedi veya cihaz bulunamadı!";
                }
            } else {
                // Kayıt Durduruluyor
                if (mediaRecorder && mediaRecorder.state !== "inactive") {
                    mediaRecorder.stop();
                    // Mikrofon kullanımını serbest bırak
                    mediaRecorder.stream.getTracks().forEach(track => track.stop());
                }
                isRecording = false;
                statusText.innerText = "Kayıt tamamlandı! Aşağıdan dinleyebilirsiniz.";
                statusText.classList.remove('text-red-500', 'font-bold');
                micIcon.className = "fa-solid fa-microphone text-2xl";
            }
        }

        // BAŞARDIM / YILDIZ GÜNCELLEME ALANI
        function toggleWordSuccess() {
            const successBtn = document.getElementById('success-btn');

            if (completedWords.has(currentWord)) {
                // Eğer zaten başardıysa ve geri almak istiyorsa
                completedWords.delete(currentWord);
                totalStars = Math.max(0, totalStars - 1);
            } else {
                // Yeni kelimeyi başardıysa
                completedWords.add(currentWord);
                totalStars++;
                
                // Konfeti fırlat (Harika Efekt!)
                triggerConfetti();
            }

            // Yıldız sayısını güncelle
            document.getElementById('star-count').innerText = totalStars;
            updateSuccessButtonState();
        }

        // Başardım Buton Görünüm Güncellemesi
        function updateSuccessButtonState() {
            const successBtn = document.getElementById('success-btn');
            const icon = document.getElementById('star-btn-icon');
            const btnText = document.getElementById('success-btn-text');

            if (completedWords.has(currentWord)) {
                // Başarılmış durum
                successBtn.className = "w-full bg-emerald-500 hover:bg-emerald-600 text-white font-extrabold p-5 rounded-2xl shadow-md flex items-center justify-center space-x-3 transition-all text-lg border-b-4 border-emerald-700";
                icon.innerText = "⭐";
                btnText.innerText = "Bu Kelimeyi Başardın! (İptal Et)";
            } else {
                // Başarılmamış varsayılan durum
                successBtn.className = "w-full bg-slate-200 hover:bg-emerald-400 hover:text-white text-slate-600 font-extrabold p-5 rounded-2xl shadow-md flex items-center justify-center space-x-3 transition-all text-lg";
                icon.innerText = "☆";
                btnText.innerText = "Bu Kelimeyi Başardım!";
            }
        }

        // TURENG 3 SANİYE BASILI TUTMA FONKSİYONLARI
        function startPress() {
            isHolding = true;
            const progressBar = document.getElementById('progress-bar');
            
            // Barı sıfırla ve animasyonlu doldur
            progressBar.className = "absolute bottom-0 left-0 h-2 bg-sky-400 w-0 holding-progress";

            pressTimer = setTimeout(function() {
                if (isHolding) {
                    // 3 Saniye tamamlandığında Tureng'i aç
                    window.open(`https://tureng.com/tr/turkce-ingilizce/${currentWord}`, '_blank');
                    cancelPress();
                }
            }, 3000);
        }

        function cancelPress() {
            isHolding = false;
            clearTimeout(pressTimer);
            const progressBar = document.getElementById('progress-bar');
            
            // Animasyonu durdur ve barı sıfırla
            progressBar.className = "absolute bottom-0 left-0 h-2 bg-sky-400 w-0";
        }

        // RENKLİ KONFETİ TETİKLEYİCİSİ
        function triggerConfetti() {
            var duration = 2.5 * 1000;
            var end = Date.now() + duration;

            (function frame() {
                confetti({
                    particleCount: 5,
                    angle: 60,
                    spread: 55,
                    origin: { x: 0, y: 0.85 }
                });
                confetti({
                    particleCount: 5,
                    angle: 120,
                    spread: 55,
                    origin: { x: 1, y: 0.85 }
                });

                if (Date.now() < end) {
                    requestAnimationFrame(frame);
                }
            }());
        }
